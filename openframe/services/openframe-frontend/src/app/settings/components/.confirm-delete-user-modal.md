A confirmation modal dialog component for confirming user deletion with destructive styling and clear warning messaging.

## Key Components

- **ConfirmDeleteUserModal** - Main modal component that displays a confirmation dialog for user deletion
- **ConfirmDeleteUserModalProps** - TypeScript interface defining the required props including open state, user name, and confirmation handler

## Usage Example

```typescript
import { useState } from 'react'
import { ConfirmDeleteUserModal } from './confirm-delete-user-modal'

function UserManagement() {
  const [showDeleteModal, setShowDeleteModal] = useState(false)
  const [selectedUser, setSelectedUser] = useState<string>('')

  const handleDeleteUser = async () => {
    try {
      // Delete user logic here
      await deleteUserAPI(selectedUser)
      setShowDeleteModal(false)
    } catch (error) {
      console.error('Failed to delete user:', error)
    }
  }

  const initiateDelete = (userName: string) => {
    setSelectedUser(userName)
    setShowDeleteModal(true)
  }

  return (
    <>
      <button onClick={() => initiateDelete('john.doe')}>
        Delete User
      </button>
      
      <ConfirmDeleteUserModal
        open={showDeleteModal}
        onOpenChange={setShowDeleteModal}
        userName={selectedUser}
        onConfirm={handleDeleteUser}
      />
    </>
  )
}
```

The modal features destructive styling with error colors, clear typography using the OpenFrame design system fonts (Azeret Mono for headings, DM Sans for body text), and proper accessibility through the AlertDialog components. The user name is highlighted in red to emphasize the destructive action.
