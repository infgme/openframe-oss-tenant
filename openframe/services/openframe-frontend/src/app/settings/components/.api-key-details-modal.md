A read-only modal component that displays comprehensive details and usage statistics for API keys in the Flamingo platform.

## Key Components

- **ApiKeyDetailsModal** - Main modal component displaying API key information including metadata, status, and usage metrics
- **ApiKeyDetailsModalProps** - Interface defining modal props including visibility state, close handler, and API key data
- **ApiKeyRecord** - Type import for API key data structure from the `use-api-keys` hook

## Usage Example

```typescript
import { ApiKeyDetailsModal } from './api-key-details-modal'
import { useApiKeys } from '../hooks/use-api-keys'

export function ApiKeyManagement() {
  const { apiKeys } = useApiKeys()
  const [selectedKey, setSelectedKey] = useState<ApiKeyRecord | null>(null)
  const [isModalOpen, setIsModalOpen] = useState(false)

  const handleViewDetails = (apiKey: ApiKeyRecord) => {
    setSelectedKey(apiKey)
    setIsModalOpen(true)
  }

  return (
    <>
      {/* Your API key list */}
      <ApiKeyDetailsModal
        isOpen={isModalOpen}
        onClose={() => setIsModalOpen(false)}
        apiKey={selectedKey}
      />
    </>
  )
}
```

The modal displays formatted date/time information, usage statistics, status indicators, and handles null values gracefully with fallback display logic.
