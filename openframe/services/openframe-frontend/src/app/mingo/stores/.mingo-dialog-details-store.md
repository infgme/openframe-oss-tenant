A Zustand store that manages the state for Mingo AI chat dialog details, including current conversation data, messages, loading states, and real-time updates.

## Key Components

- **useMingoDialogDetailsStore** - Main store hook providing dialog state management
- **State Management** - Tracks current dialog, admin messages, loading states, and pagination
- **Real-time Support** - Handles live message updates and typing indicators
- **Message Management** - Provides functions for adding, filtering, and deduplicating messages
- **Error Handling** - Separate error states for dialog and message operations

## Usage Example

```typescript
import { useMingoDialogDetailsStore } from './mingo-dialog-details-store'

function ChatDialog() {
  const {
    currentDialog,
    adminMessages,
    isLoadingMessages,
    setCurrentDialogId,
    addRealtimeMessage,
    setTypingIndicator
  } = useMingoDialogDetailsStore()

  // Load a specific dialog
  const loadDialog = (dialogId: string) => {
    setCurrentDialogId(dialogId)
  }

  // Handle incoming real-time message
  const handleNewMessage = (message: Message) => {
    addRealtimeMessage(message)
  }

  // Show typing indicator
  const handleTyping = () => {
    setTypingIndicator(true)
  }

  return (
    <div>
      {currentDialog && (
        <div>
          <h2>{currentDialog.title}</h2>
          {adminMessages.map(msg => (
            <div key={msg.id}>{msg.content}</div>
          ))}
          {isLoadingMessages && <div>Loading...</div>}
        </div>
      )}
    </div>
  )
}
```
